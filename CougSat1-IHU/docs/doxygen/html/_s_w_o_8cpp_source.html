103 void SWO_Setup(void) {
104  /* Enable trace subsystem (we'll use ITM and TPIU) */
105  SCS_DEMCR |= SCS_DEMCR_TRCENA;
106 
107  /* Use Manchester code for asynchronous transmission */
108  TPIU_SPPR = TPIU_SPPR_ASYNC_MANCHESTER;
109  TPIU_ACPR = 7;
110 
111  /* Data width is 1 byte */
112  TPIU_CSPSR = TPIU_CSPSR_BYTE;
113 
114  /* Formatter and flush control */
115  TPIU_FFCR &= ~TPIU_FFCR_ENFCONT;
116 
117  /* Enable TRACESWO pin for async mode */
118  DBGMCU_CR = DBGMCU_CR_TRACE_IOEN | DBGMCU_CR_TRACE_MODE_ASYNC;
119 
120  /* Unlock access to ITM registers */
121  /* FIXME: Magic numbers... Is this Cortex-M3 generic? */
122  *((volatile uint32_t*)0xE0000FB0) = 0xC5ACCE55;
123 
124  /* Enable ITM with ID = 1 */
125  ITM_TCR = (1 << 16) | ITM_TCR_ITMENA;
126  /* Enable stimulus port 1 */
127  ITM_TER[0] = 1;
128 }

136 void SWO_Setup_1(void) {
137  U32 SWOPrescaler;
138  U32 SWOSpeed;
139 
140 //<Init PLL, set CPU clock to 72 MHz> // Optional, so I do not pos it here
141 
142  SWOSpeed = 6000000;
143  *((volatile unsigned *)0xE000EDFC) = 0x01000000; // "Debug Exception and Monitor Control Register (DEMCR)"
144  *((volatile unsigned *)0xE0042004) = 0x00000027;
145  *((volatile unsigned *)0xE00400F0) = 0x00000002; // "Selected PIN Protocol Register": Select which protocol to use for trace output (2: SWO)
146  SWOPrescaler = (72000000 / SWOSpeed) - 1; // SWOSpeed in Hz
147  *((volatile unsigned *)0xE0040010) = SWOPrescaler; // "Async Clock Prescaler Register". Scale the baud rate of the asynchronous output
148  *((volatile unsigned *)0xE0000FB0) = 0xC5ACCE55; // ITM Lock Access Register, C5ACCE55 enables more write access to Control Register 0xE00 :: 0xFFC
149  *((volatile unsigned *)0xE0000E80) = 0x0001000D; // ITM Trace Control Register
150  *((volatile unsigned *)0xE0000E40) = 0x0000000F; // ITM Trace Privilege Register
151  *((volatile unsigned *)0xE0000E00) = 0x00000001; // ITM Trace Enable Register. Enabled tracing on stimulus ports. One bit per stimulus port.
152  *((volatile unsigned *)0xE0001000) = 0x400003FE; // DWT_CTRL
153  *((volatile unsigned *)0xE0040304) = 0x00000100; // Formatter and Flush Control Register
154 }

168 void SWO_PrintChar(char c) {
169 
170 #if(1)
171 //Use CMSIS_core_DebugFunctions. See core_cm3.h
172  ITM_SendChar (c);
173 
174 #else
175 //Use Segger example. Basically same as CMSIS
176 
177  // Check if ITM_TCR.ITMENA is set
178  if ((ITM_TCR & 1) == 0) {
179  return;
180  }
181 
182  // Check if stimulus port is enabled
183  if ((ITM_ENA & 1) == 0) {
184  return;
185  }
186 
187  // Wait until STIMx FIFO is ready, then send data
188 // while ((ITM_STIM_U8(0) & 1) == 0);
189  while (!(ITM_STIM_U8(0) & ITM_STIM_FIFOREADY));
190  ITM_STIM_U8(0) = c;
191 
192 // while ((ITM_STIM_U32(0) & 1) == 0);
193 // ITM_STIM_U32(0) = c;
194 
195 #endif
196 }
