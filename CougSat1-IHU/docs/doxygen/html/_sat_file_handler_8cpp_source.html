94 void SatFileHandler::check()
95 {
96  uint8_t full[BLOCKSIZE];
97  std::fill_n(full, BLOCKSIZE, 0xFF);
98  uint8_t blockData[BLOCKSIZE] = {0}, tmp[BLOCKSIZE] = {0};
99  bd_size_t numBlocks = sdbd->size() / BLOCKSIZE;
100  uint32_t numBad = 0;
101 
102  if (debug)
103  pc->printf("SD Block Count: %lu\r\n", numBlocks);
104 
105  std::chrono::high_resolution_clock::time_point t1
106  = std::chrono::high_resolution_clock::now();
107 
108  auto checkBlock = [&](bd_size_t i) {
109  sdbd->read(tmp, i, BLOCKSIZE);
110  sdbd->program(full, i, BLOCKSIZE);
111  sdbd->read(blockData, i, BLOCKSIZE);
112  if (!compareArrays(blockData, full, BLOCKSIZE)) {
113  numBad++;
114  }
115  sdbd->program(&tmp, i, BLOCKSIZE);
116  };
117 
118  for (bd_size_t i = 0; i < numBlocks; i += BLOCKSIZE) {
119  checkBlock(i);
120  }
121 
122  std::chrono::high_resolution_clock::time_point t2
123  = std::chrono::high_resolution_clock::now();
124  std::chrono::seconds dur
125  = std::chrono::duration_cast<std::chrono::seconds>(t2 - t1);
126 
127  int32_t seconds = static_cast<int64_t>(dur.count());
128  pc->printf("Check took %ld seconds\r\n", seconds);
129  pc->printf("Block device check: Done\r\n");
130  pc->printf("Number of bad blocks: %lu\r\n", numBad);
131 }
